Grammar

The EPL grammar is specified in the source file EplParser.jjt. The following diagrams illustrate the grammar.

Nonterminals

{@link ASTS S} ::= ( {@link ASTStmt Stmt} ";" )+

{@link ASTStmt Stmt} ::= {@link ASTPrint Print}
       | {@link ASTExecute Execute}
       | {@link ASTForeach Foreach}
       | {@link ASTIf If}
       | {@link ASTListener Listener}
       | {@link ASTBroker Broker}
       | {@link ASTCondition Condition}
       | {@link ASTAction Action}
       | {@link ASTSituation Situation}

{@link ASTPrint Print} ::= "print" ( {@link ASTPrintCatenation PrintCatenation} )+

{@link ASTPrintCatenation PrintCatenation} ::= {@link ASTPrintExpr PrintExpr} [ ( "+" {@link ASTPrintExpr PrintExpr} )+ ]

{@link ASTPrintExpr PrintExpr} ::= {@link ASTVarExpr VarExpr}
            | <STRING>

{@link ASTVarExpr VarExpr} ::= <ID> [ "." <ID> ]

{@link ASTExecute Execute} ::= "execute" <ID> "as" <STRING> [ {@link ASTParams Params} ]

{@link ASTParams Params} ::= ( {@link ASTParam Param} )+

{@link ASTParam Param} ::= {@link ASTVarExpr VarExpr}
        | <STRING>

{@link ASTForeach Foreach} ::= "foreach" <ID> "in" {@link ASTVarExpr VarExpr} {@link ASTBlock Block}

{@link ASTBlock Block} ::= "{" ( {@link ASTStmt Stmt} ";" )+ "}"

{@link ASTIf If} ::= "if" {@link ASTExpr Expr} {@link ASTBlock Block}

{@link ASTListener Listener} ::= "listener" <ID> "for" <STRING> "in" <ID>

{@link ASTBroker Broker} ::= "broker" <ID> "is" <STRING>

{@link ASTCondition Condition} ::= "condition" <ID> "is" {@link ASTExpr Expr} "within" <DIGITS>

{@link ASTExpr Expr} ::= {@link ASTVarExpr VarExpr} [ {@link ASTConditional Conditional} ]
       | {@link ASTNotConditional NotConditional}
       | "(" {@link ASTExpr Expr} ")"

{@link ASTConditional Conditional} ::= {@link ASTOrConditional OrConditional}
              | {@link ASTAndConditional AndConditional}

{@link ASTNotConditional NotConditional} ::= "not" {@link ASTExpr Expr}

{@link ASTOrConditional OrConditional} ::= "or" {@link ASTExpr Expr}

{@link ASTAndConditional AndConditional} ::= "and" {@link ASTExpr Expr}

{@link ASTAction Action} ::= "action" <ID> "is" {@link ASTBlock Block}

{@link ASTSituation Situation} ::= "when" <ID> "do" {@link ASTList List}

{@link ASTList List} ::= {@link ASTVarExpr VarExpr} [ ( "," {@link ASTVarExpr VarExpr} )+ ]

Terminals

 ID: ["A"-"Z","a"-"z"] ( ["A"-"Z","a"-"z","0"-"9"] )*

 STRING: "\"" (~["\""])* "\""

 DIGITS: ["1"-"9"] ( ["0"-"9"] )*
